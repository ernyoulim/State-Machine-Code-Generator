static void doStep_Steuerung_Region_Nachtbetrieb(uint32_t tick) {

        #ifdef USESOCKET

            #endif // USESOCKET

static void doStep_Steuerung_Region1(uint32_t tick) {
    switch(state_Steuerung_Region1) {
    case S_DEFAULT:
        declareTimer(1,2,TimeOutRegion1Ev);
        startTimer(1,tick);
        state_Steuerung_Region1 = S_Steuerung_Region1_Released;
        #ifdef USESOCKET
        sendActivation("Controller/Steuerung/Region1/Released");
        #endif // USESOCKET
        break;
    case S_Steuerung_Region1_Released:
        if (eventIsSet(TimeOutRegion1Ev)){
            if (port_read_Controller_Taster()==1){
                setEvent(TasterEv);
                declareTimer(1,2,TimeOutRegion1Ev);
                startTimer(1,tick);
                state_Steuerung_Region1 = S_Steuerung_Region1_pressed;
                #ifdef USESOCKET
                sendDeactivation("Controller/Steuerung/Region1/Released");
                sendActivation("Controller/Steuerung/Region1/pressed");
                #endif // USESOCKET
                break;
            } else {
                declareTimer(1,2,TimeOutRegion1Ev);
                startTimer(1,tick);
            }
        }
        break;
    case S_Steuerung_Region1_pressed:
        if (eventIsSet(TimeOutRegion1Ev)){
            if (port_read_Controller_Taster()==0){
                declareTimer(1,2,TimeOutRegion1Ev);
                startTimer(1,tick);
                state_Steuerung_Region1 = S_Steuerung_Region1_Released;
                #ifdef USESOCKET
                sendDeactivation("Controller/Steuerung/Region1/pressed");
                sendActivation("Controller/Steuerung/Region1/Released");
                #endif // USESOCKET
                break;
            } else {
                declareTimer(1,2,TimeOutRegion1Ev);
                startTimer(1,tick);
            }
        }
        break;
    }
}

static void leave_Steuerung_Region1() {
    #ifdef USESOCKET
    switch(state_Steuerung_Region1) {
    case S_Steuerung_Region1_Released:
        sendDeactivation("Controller/Steuerung/Region1/Released");
        break;
    case S_Steuerung_Region1_pressed:
        sendDeactivation("Controller/Steuerung/Region1/pressed");
        break;
    }
    #endif // USESOCKET
    cancelTimer(1);
    state_Steuerung_Region1 = S_DEFAULT;
}


static void doStep_Steuerung_Region(uint32_t tick) {
    switch(state_Steuerung_Region) {
    case S_DEFAULT:
        state_Steuerung_Region = S_Steuerung_Region_Tagbetrieb;
        #ifdef USESOCKET
        sendActivation("Controller/Steuerung/Region/Tagbetrieb");
        #endif // USESOCKET
        doStep_Steuerung_Region_Tagbetrieb(tick);
        break;
    case S_Steuerung_Region_Tagbetrieb:
        if (eventIsSet(TasterEv)){
            port_write_Controller_Rot(0);
            port_write_Controller_Gelb(1);
            port_write_Controller_Gruen(0);
            cancelTimer(0);
            state_Steuerung_Region = S_Steuerung_Region_Nachtbetrieb;
            #ifdef USESOCKET
            sendActivation("Controller/Steuerung/Region/Nachtbetrieb");
            sendDeactivation("Controller/Steuerung/Region/Tagbetrieb");
            #endif // USESOCKET
            leave_Steuerung_Region_Tagbetrieb();
            doStep_Steuerung_Region_Nachtbetrieb(tick);
        }
        else {
            doStep_Steuerung_Region_Tagbetrieb(tick);
        }
        break;
    case S_Steuerung_Region_Nachtbetrieb:
        if (eventIsSet(TasterEv)){
            cancelTimer(0);
            state_Steuerung_Region = S_Steuerung_Region_Tagbetrieb;
            #ifdef USESOCKET
            sendActivation("Controller/Steuerung/Region/Tagbetrieb");
            sendDeactivation("Controller/Steuerung/Region/Nachtbetrieb");
            #endif // USESOCKET
            leave_Steuerung_Region_Nachtbetrieb();
            doStep_Steuerung_Region_Tagbetrieb(tick);
        }
        else {
            doStep_Steuerung_Region_Nachtbetrieb(tick);
        }
        break;
    }
}

static void leave_Steuerung_Region() {

    switch(state_Steuerung_Region) {
    case S_Steuerung_Region_Tagbetrieb:
        leave_Steuerung_Region_Tagbetrieb();
        #ifdef USESOCKET
        sendDeactivation("Controller/Steuerung/Region/Tagbetrieb");
        #endif // USESOCKET
        break;
    case S_Steuerung_Region_Nachtbetrieb:
        leave_Steuerung_Region_Nachtbetrieb();
        #ifdef USESOCKET
        sendDeactivation("Controller/Steuerung/Region/Nachtbetrieb");
        #endif // USESOCKET
        break;
    }
    state_Steuerung_Region = S_DEFAULT;
}


static void doStep_Steuerung(uint32_t tick) {
    doStep_Steuerung_Region(tick);
    doStep_Steuerung_Region1(tick);
}

static void leave_Steuerung() {
    leave_Steuerung_Region();
    leave_Steuerung_Region1();
    #ifdef USESOCKET
    sendDeactivation("Controller/Steuerung");
    #endif // USESOCKET
}


static void Controller_stmDoStep(uint32_t tick) {
   switch (state) {
   case S_DEFAULT:
        #ifdef USESOCKET
        sendActivation("Controller/Steuerung");
        #endif // USESOCKET
        state = S_Steuerung;
        doStep_Steuerung(tick);
        break;
   case S_Steuerung:
        doStep_Steuerung(tick);
        break;
   }
}

}
